    public void generateaccountnumber() {
        String path = "jdbc:mysql://localhost/";
        String place = "bankingdb";
        try {
            try {
                Class.forName("com.mysql.jdbc.Driver");
                Connection myconnection = (Connection) DriverManager.getConnection(path + place, "root", "");
                String q = "Select accountno from customer_info order by accountno DESC";
                PreparedStatement mystatement = (PreparedStatement) myconnection.prepareStatement(q);
                ResultSet myresult = mystatement.executeQuery();
                if (myresult.next()) {
                    long oldno = Long.parseLong(myresult.getString("accountno"));
                    long newno = oldno + 1;
                    l4.setText("" + newno);
                } else {
                    l4.setText("1103090278020001");
                }
                java.util.Date obj = new java.util.Date();
                SimpleDateFormat dateformat = new SimpleDateFormat("yyy-MM-dd");

            }

            JOptionPane.showMessageDialog(rootPane, "Error in Connection " + e.getMessage());
        }
    }

    private void btn1ActionPerformed(java.awt.event.ActionEvent evt) {                                     
        // TODO add your handling code here:
      
        fc = new JFileChooser();

        fc.setFileFilter(new FileFilter() {

            @Override
            public boolean accept(File f) {
                if (f.isDirectory()) {
                    return true;
                }
                int pos = f.getName().lastIndexOf('.');
                if (pos == -1) {
                    return false;
                } else {
                    String extention = f.getName().substring(pos + 1);//saving the extension
                    String str[] = {"gif", "png", "jpg", "jpeg"};//allowed extentions
                    for (String allwd : str) {

                        if (extention.equalsIgnoreCase(allwd)) {
                            return true;
                        }
                    }
                    return false;
                }

            }

            @Override
            public String getDescription() {

                return "";
            }
        });
        int result = fc.showOpenDialog(null);

        if (result == JFileChooser.APPROVE_OPTION) {
            myfile = fc.getSelectedFile();

            try {
                l2.setIcon(new ImageIcon(ImageIO.read(myfile)));
            } catch (IOException e) {
                JOptionPane.showMessageDialog(rootPane, "Error");
                e.printStackTrace();
            }
        }
    }
